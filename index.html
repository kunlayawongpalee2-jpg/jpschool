function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheets()[0]; const data = JSON.parse(e.postData.contents); // กำหนดลำดับข้อมูล A-I ให้ตรงกับที่คุณครูแจ้งเป๊ะๆ const values = [ new Date(), // A: วันที่บันทึก data.academicYear, // B: ปีการศึกษา data.teacher, // C: ชื่อ-นามสกุลคุณครู data.level, // D: ระดับชั้น data.name, // E: ชื่อหลักสูตรที่อบรม data.location, // F: สถานที่ data.organizer, // G: หน่วยงานที่จัด data.hours, // H: จำนวนชั่วโมง data.date // I: วันที่อบรม ]; if (data.action === "add") { sheet.appendRow(values); } else if (data.action === "edit") { const row = data.rowId; sheet.getRange(row, 1, 1, values.length).setValues([values]); } return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } function doGet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheets()[0]; const rows = sheet.getDataRange().getValues(); const headers = rows.shift(); // ดึงหัวตารางออก // ส่งข้อมูลกลับไปแบบระบุชื่อช่อง (จะได้ไม่สลับกัน) const result = rows.map((r, index) => { return { rowId: index + 2, timestamp: r[0], academicYear: r[1], teacher: r[2], level: r[3], name: r[4], location: r[5], organizer: r[6], hours: r[7], date: r[8] }; }); return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); }
